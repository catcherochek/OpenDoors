SELECT COUNT(*) as count FROM (SELECT COUNT(*) FROM oc_product p LEFT JOIN oc_product_description pd ON (p.product_id = pd.product_id AND pd.language_id = '1') LEFT JOIN oc_product_to_store p2s ON (p.product_id = p2s.product_id AND p2s.store_id = '0') WHERE p2s.store_id = '0'  GROUP BY p.product_id) as count_table
